#include <iostream>
#include <string.h>

using namespace std;

int getPeriod(string str)
{
    int str_len = str.length();
    for( int i = 1; i <= str_len/2; i++ )
    {
        if( str_len % i == 0 )
        {
            string sub = str.substr(0,i);
            int period = str_len / sub.length();
            for( int j = 1; j < period; j++ ){
                string temp = str.substr(i*j,i);
                if( sub != temp ){
                    period = 0;
                    break;
                }
            }
            if( period != 0 )
                return period;
        }
    }
    return 0;
}

int main()
{
    int len;
    int times = 0;
    cin >> len;
    while(len != 0 )
    {
        times++;
        string input;
        cin >> input;
        cout << "Test case #" << times << endl;
        for( int i = 2; i <= len; i++ )
        {
            string sub = input.substr(0,i);
            int period = getPeriod(sub);
            if( period > 0 )
            {
                cout << sub.length() << " " << period << endl;
            }
        }
        cout << endl;
        cin >> len;
    }
    return 0;
}
